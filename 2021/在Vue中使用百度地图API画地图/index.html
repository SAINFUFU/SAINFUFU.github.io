在Vue中使用百度地图API画地图

2021-02-20
CS-Vue

安装

# 安装echarts
yarn add echarts

引入百度api

在public/index.html中引入

<script type="text/javascript" src="http://api.map.baidu.com/api?v=3.0&ak=x768hLYscNh3RdGVr4emCqyQMkBG8Vs6"></script>

在main.js中导入echarts和bmap扩展

import * as echarts from 'echarts'
import "echarts/extension/bmap/bmap"
Vue.prototype.$echarts = echarts

在Vue中使用

<div ref="myMap" style="height:400px;"></div>

let option = {
    // 加载 bmap 组件
    bmap: {
        // 百度地图中心经纬度
        center: [120.13066322374, 30.240018034923],
        // 百度地图缩放
        zoom: 5,
        // 是否开启拖拽缩放，可以只设置 'scale' 或者 'move'
        roam: true,
        // 百度地图的自定义样式，见 http://developer.baidu.com/map/jsdevelop-11.htm
        mapStyleV2: {
            styleId: "7f71a451bcb4c4bde7b2110fe5dd0fdf"
        }
    },
    series: [{
        type: 'scatter',
        // 使用百度地图坐标系
        coordinateSystem: 'bmap',
        // 数据格式跟在 geo 坐标系上一样，每一项都是 [经度，纬度，数值大小，其它维度...]
        data: [ [120, 30, 1] ]
    }]
  }
  let chart = this.$echarts.init(this.$refs.myMap)
  chart.setOption(option)
