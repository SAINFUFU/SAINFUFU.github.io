算m点问题

2020-05-08
ACM-回溯法

题目描述

算m点问题。给定k个正数，用算术运算符+、-、*、/进行连接，使得最后的结果恰为m。

分析

这题需要使用回溯法，对所有可能的顺序进行深度优先遍历，直到找到结果为m的路径，进行输出。
由于本题的意思是可以加上括号的那种，因此两个相邻的数之间一共有6种情况。
以a，b分别代表两个数，可以表示为
a+b,a-b,b-a,a*b,a/b,b/a
因此，将前两个数运算得到的结果value，与选择的下一个数进行上述六种之一的运算，就可以实现加括号的效果。

由于节点数过多，还需要剪枝算法，把肯定不可能的情况舍弃，不进行扩展。
这题遍历并不难，关键在如何优化剪枝函数。由于在没找到结果之前，不能轻易判断该路径的正确性，因此就用下面两个情况进行剪枝。

① 因为只需要输出一个正确解，所以在找到一个输出结果之后，直接退出搜索，不再扩展其他子节点。
② 由于会出现重复数的情况，因此在同一层的剩余节点选数的时候，如果出现相同的值，就跳过，避免出现重复操作，浪费时间。实现方式为每一层都设一个数组记录是否使用过相同的值。

时间复杂度分析：
最坏情况：需要遍历所有的排列，一开始进入搜索时选择初始值有k种情况，接着有6(k-1)中情况到下一层，依次类推，最终的路径总数为`6^(n-1)n!。 因此该算法的时间复杂度为：O(6^(n-1)n!)`

代码

#include <iostream>
#include <cstdio>
#include <string.h>
#include <algorithm>
#include <time.h>

using namespace std;
const int N = 10000;
int k,m,flag;//flag用于标记是否已经找到一组解
int vis[N][N];//标记该数是否已经用过
double a[N];//存放k个数的顺序
char op[N];//记录运算符
void print()//输出结果
{
    double res = a[0];
    for(int i = 1;i<k;i++)
    {
        a[i-1] = res;//覆盖前一个值，便于进行循环操作
        switch(op[i])//计算两个的值,并输出
        {
            //如果出现负数的情况，加括号
            case '+':res = a[i-1]+a[i];
                     cout<<(a[i-1]<0?"(":"")<<a[i-1]<<(a[i-1]<0?")":"")<<op[i]<<(a[i]<0?"(":"")<<a[i]<<(a[i]<0?")":"")<<"="<<res<<";";break;
            case '-':res = a[i-1]-a[i];
                     cout<<(a[i-1]<0?"(":"")<<a[i-1]<<(a[i-1]<0?")":"")<<op[i]<<(a[i]<0?"(":"")<<a[i]<<(a[i]<0?")":"")<<"="<<res<<";";break;
            case '_':res = a[i]-a[i-1];
                     cout<<(a[i]<0?"(":"")<<a[i]<<(a[i]<0?")":"")<<'-'<<(a[i-1]<0?"(":"")<<a[i-1]<<(a[i-1]<0?")":"")<<"="<<res<<";";break;
            case '*':res = a[i-1]*a[i];
                     cout<<(a[i-1]<0?"(":"")<<a[i-1]<<(a[i-1]<0?")":"")<<op[i]<<(a[i]<0?"(":"")<<a[i]<<(a[i]<0?")":"")<<"="<<res<<";";break;
            case '/':res = a[i-1]/a[i];
                     cout<<(a[i-1]<0?"(":"")<<a[i-1]<<(a[i-1]<0?")":"")<<op[i]<<(a[i]<0?"(":"")<<a[i]<<(a[i]<0?")":"")<<"="<<res<<";";break;
            case '\\':res = a[i]/a[i=1];
                      cout<<(a[i]<0?"(":"")<<a[i]<<(a[i]<0?")":"")<<'/'<<(a[i-1]<0?"(":"")<<a[i-1]<<(a[i-1]<0?")":"")<<"="<<res<<";";break;
        }
    }
    cout<<endl;
}
void dfs(int step,double value)
{
    if(flag) return;//找到解，将之后将要搜索的节点直接退出

    // if(!flag && step == k && int((value-m)*10000000)==0)//求时间复杂度时使用，同时将上下两句注释掉
    if(step == k && int((value-m)*10000000)==0)//如果找到就输出
    {
        print();//输出结果
        flag = 1;//找到解，置1，剪枝
        return;//退出
    }
    for(int i = 0;i<k;i++)
        vis[step][int(a[i])] = 0;//初始化
    for(int i = step;i<k;i++)//从后面没有确定的数中选一个
    {
        if(!vis[step][int(a[i])])//如果将要扩展的节点已经扩展过，只是重复操作，剪枝
        {
            vis[step][int(a[i])] = 1;
            swap(a[i],a[step]);//交换
            op[step] = '+';//加
            dfs(step+1,value+a[step]);
            op[step] = '-';//减
            dfs(step+1,value-a[step]);
            op[step] = '_';//反减
            dfs(step+1,a[step]-value);
            op[step] = '*';//乘
            dfs(step+1,value*a[step]);
            op[step] = '/';//除
            dfs(step+1,value/a[step]);
            if(value != 0)
            {
                op[step] = '\\';//反除
                dfs(step+1,a[step]/value);
            }
            swap(a[i],a[step]);//还原
        }
    }
}
void solve()
{
    flag = 0;//初始置0
    for(int i = 0;i<k&&!flag;i++)//遍历第一个数的值
    {
        swap(a[i],a[0]);//交换
        dfs(1,a[0]);//进入搜索
        swap(a[i],a[0]);//还原
    }    
    if(!flag) printf("No Solution!\n");//找不到输出"No Solution!"
}
int main()
{
    freopen("input.txt","r",stdin);
    freopen("output.txt","w",stdout);
    while(cin>>k>>m)
    {
        for(int i = 0;i<k;i++)
        {
            scanf("%lf",&a[i]);
        }
        clock_t start,end;
        start = clock();//开始计时
        solve();
        end = clock();//结束计时
        printf("k=%d,time=%d\n\n",k,end-start);
    }
    return 0;
}

总结

本次实验在算法设计上只是剪枝函数的确定以及算法的优化上需要好好思考一下。
回溯法基本上就是深度优先搜索+剪枝函数。与枚举的思想类似，把所有能走的路都走一遍，直到找到符合要求的目标节点。这种算法的时间复杂度都很高，只能进行小规模的运算。就像此题，当数据规模为8时，就耗费了36秒。

在计算时间的时候，能找到解的情况与找不到解的情况下，时间可能会有较大的差异，使时间复杂度曲线不是很合理。因此我采用了每组数据都需要全部遍历完的方法，获取正确的时间复杂度。
其实本来想跑出k=9时的时间的，奈何实在是太慢了，得有好几分钟了，就只算到了k=8。这个经历，让我更加明白了，学好算法有多重要，暴力搜索效率实在太低了。

其实我在一开始设计好算法，写完程序后，参考了一下参考答案。我惊奇地发现我这个程序的逻辑上有两点问题。

  1. 我在扩展子节点的时候，只考虑了4种情况（即+-*/）,忽略了反减和反除的情况。我仔细思考了一下为什么参考答案上要计算6种，而不是4种。后来发现进行-/位置的倒换，其实意味着这个题目允许加括号，而不只是限于普通的四则运算。

    可以以4 24 1 5 5 5该数据为例
4种情况：没有解
6种情况：有解   1/5=0.2;5-0.2=4.8;4.8*5=24;

    我又琢磨了一下书上的题目和给的样例，将算式写成最简形式。即(7*3*12-2)/2=125
    这不是妥妥的可以加括号么！
    于是，按照这个思想，我对原先的算法进行了修改。成功得出了正确的结果。

  2. 在输出上出现一些纰漏，没有考虑负数的情况，于是进行了修正，对负数加上括号。
