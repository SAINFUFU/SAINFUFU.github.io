代码

template<class Type>
class Traveling
{
    friend int main();
    public:
        Type BBTSP(int v[]);
    private:
        int n,*bestp;//增加当前最优排列数组bestp
        Type **a,NoEdge,c,bestc;
};
template<class Type>
Type Traveling<Type>::BBTSP(int v[])
{
    MinHeap<MinHeapNode<Type> >H(1000);//定义最小堆容量为1000
    Type *MinOut = new Type [n+1];
    //计算MinOut[i]=顶点i的最小出边费用
    Type MinSum = 0;
    for(int i = 1;i<=n;i++)
    {
        Type Min = NoEdge;
        for(int j = 1;j<=n;j++)
            if(a[i][j] != NoEdge && (a[i][j] <Min ||Min == NoEdge))
                Min = a[i][j];
        if(Min == NoEdge)//无回路
            return NoEdge;
        MinOut[i] = Min;
        MinSum += Min;
    }
    //初始化
    MinHeapNode<Type> E;
    E.x = new int [n];
    for(int i = 0;i<n;i++)
        E.x[i] = i+1;
    E.s = 0;
    E.cc = 0;
    E.rcost = MinSum;
    Type bestc = NoEdge;
    //搜索排列空间树
    while(E.s < n-1 && E.lcost< bestc)
    {
        if(E.s == n-2)
        {
            //如果当前有回路 且 费用小于当前最优解
            if(a[E.x[n-2]][E.x[n-1]] !=NoEdge && a[E.x[n-1]][E.x[1]] !=NoEdge && (E.cc+a[E.x[n-2]][E.x[n-1]]+a[E.x[n-1]][E.x[1]]<bestc || bestc == NoEdge))
            {
                bestc = E.cc+a[E.x[n-2]][E.x[n-1]]+a[E.x[n-1]][E.x[1]];
                for(int j = 0;j<n;j++)
                    bestp[j] = E.x[j];//存储目前最优排列
            }
            else 
                delete[] E.x //舍弃扩展节点
        }
        else
        {
            for(int i = E.s+1;i<n;i++)
            {
                if(a[E.x[E.s]][E.x[i]] !=NoEdge)//扩展可行的节点
                {
                    Type cc = E.cc+a[E.x[E.s]][E.x[i]];
                    Type rcost = E.rcost + MinOut[E.x[E.s]];
                    Type b = cc + rcost;
                    if(b<bestc || bestc == NoEdge)//如果包含最优解
                    {
                        MinHeapNode<Type> N;
                        N.x = new int [n];
                        for(int j = 0;j<n;j++)
                            N.x[j] = E.x[j];
                        N.x[E.s+1] = E.x[i];
                        N.x[i] = E.x[E.s+1];
                        N.cc = cc;
                        N.s = E.s+1;
                        N.lcost = b;
                        N.rcost = rcost;
                        H.Insert(N);
                    }
                    delete[] E.x;
                }
                try{
                    H.DeleteMin(E);//取下一扩展节点
                }
                catch(OutOfBound){
                    break;//如果堆空，就退出
                }
            }
        }  
    }
    if(bestc == NoEdge)
        return NoEdge;
    for(int i = 0;i<n;i++)
        v[i+1] = bestp[i];
    while(1)//释放所有节点
    {
        delete[] E.x;
        try{H.DeleteMin(E);}
        catch(OutOfBounds){break;}
    }
    return bestc;
}
