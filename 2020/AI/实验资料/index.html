安装fusibile时步骤：

git clone https://github.com/YoYo000/fusibile
cd fusibile
mkdir build
cd build
cmake ..
make

python colmap2mvsnet.py –dense_folder ./data/dense_folder –max_d 192 –interval_scale 1.06

python test.py –dense_folder ./data/scan3/ –regularization ‘3DCNNs’ –max_w 544 –max_h 416 –max_d 192 –interval_scale 1.06

python depthfusion.py –dense_folder ./data/scan3/ –fusibile_exe_path ./fusibile/build/fusibile –prob_threshold 0.3
